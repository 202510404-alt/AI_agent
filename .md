from pathlib import Path

# 🎯 [수정 1] switch.py 연동 및 마스터 토글 콘솔 모드 동기화
# 원래는 switch.py와 연동되지 않아 프로젝트 단위 스캔을 인지하지 못했습니다.
# ======================================================================
try:
    from cline_tools.switch import SCAN_MODE
except ImportError:
    SCAN_MODE = "ROOT"

# ======================================================================
# 🎯 [경로 방어선 절대 고정]
# 현재 스크립트 파일(create_ai_map.py)의 위치는 무조건 프로젝트루트/cline_tools 입니다.
# 따라서 .parent는 cline_tools가 되고, 그 .parent가 진짜 프로젝트 마스터 루트입니다.
# ======================================================================
SCRIPT_DIR = Path(__file__).parent.resolve()
PROJECT_ROOT = SCRIPT_DIR.parent  # 🔥 상위의 진짜 프로젝트 루트 강제 추적

# 🚨 모든 로드 및 저장 경로를 격리 보관소 폴더 구조로 완전 강제 리다이렉트
OUTPUT_FILE_PATH = PROJECT_ROOT / "system_maps" / "AI_CODEBASE_MAP.md"
REGISTRY_JSON_PATH = PROJECT_ROOT / "system_memory" / "registry_constants.json"
PROTOCOL_JSON_PATH = PROJECT_ROOT / "system_memory" / "data_protocols.json"

def parse_python_file(file_path: Path):
    """[형님 원본 규격 100% 완벽 보존] 실시간으로 라인 범위, 클래스/함수, 임포트를 징집합니다."""
    compact_symbols_info = []
    imports = []
    global_vars = []
    
    try:
        with open(file_path, "r", encoding="utf-8") as f:
            node = ast.parse(f.read(), filename=str(file_path))
            
        for child in node.body:
            # 1. 임포트 모듈 추적
            if isinstance(child, (ast.Import, ast.ImportFrom)):
                if isinstance(child, ast.Import):
                    for n in child.names:
                        imports.append(n.name)
                else:
                    mod = child.module if child.module else ""
                    for n in child.names:
                        imports.append(f"{mod}.{n.name}" if mod else n.name)
            
            # 2. 전역 변수(상수) 목록 추적
            elif isinstance(child, ast.Assign):
                for target in child.targets:
                    if isinstance(target, ast.Name):
                        global_vars.append(target.id)
                        
            # 3. 최상위 클래스 정의 및 스켈레톤 추출
            elif isinstance(child, ast.ClassDef):
                start_ln = child.lineno
                end_ln = getattr(child, "end_lineno", start_ln)
                methods = [n.name for n in child.body if isinstance(n, ast.FunctionDef)]
                compact_symbols_info.append({
                    "type": "class",
                    "name": child.name,
                    "range": [start_ln, end_ln],
                    "details": f"Methods: {methods}" if methods else "Empty Class"
                })
                
            # 4. 최상위 독립 함수 정의 추출
            elif isinstance(child, ast.FunctionDef):
                start_ln = child.lineno
                end_ln = getattr(child, "end_lineno", start_ln)
                compact_symbols_info.append({
                    "type": "function",
                    "name": child.name,
                    "range": [start_ln, end_ln],
                    "details": f"Args: {[a.arg for a in child.args.args]}"
                })
    except Exception:
        pass
        
    return compact_symbols_info, imports, global_vars

def generate_codebase_map(target_dir: Path = None):
    """인간용 일반 마스터 지도를 굽는 함수"""
    # ======================================================================
    # 🎯 [수정 2] SCAN_MODE 스위치에 맞춰 수색 대상(scan_target) 유연화
    # 원래는 scan_target = Path("src")로 고정되어 미아가 되는 현상을 해결했습니다.
    # ======================================================================
    if target_dir is None:
        if SCAN_MODE == "ROOT":
            scan_target = PROJECT_ROOT
        else:
            scan_target = PROJECT_ROOT / "src"
    else:
        scan_target = target_dir

    if not scan_target.exists():
        print(f"⚠️ {scan_target} 경로가 존재하지 않아 일반 지도 출력을 스킵합니다.")
        return

    target_files = []
    excludes = [".venv", ".git", "__pycache__", "system_memory", "system_maps"]
    
    for root, dirs, files in os.walk(scan_target, followlinks=True):
        # 🎯 [수정 3] 경로 구분자 리눅스식(/) 정규화 적용
        normalized_root = root.replace("\\", "/")
        if any(kw in normalized_root for kw in excludes):
            continue
        for file in files:
            if file.endswith(".py"):
                target_files.append(Path(root) / file)

    HUMAN_MAP_PATH = PROJECT_ROOT / "system_maps" / "CODEBASE_MAP.md"
    os.makedirs(HUMAN_MAP_PATH.parent, exist_ok=True)
    
    with open(HUMAN_MAP_PATH, "w", encoding="utf-8") as f:
        f.write("# 🏗️ 짭커서 프로젝트 CODEBASE MAP\n\n")
        f.write(f"현재 인덱싱된 총 파일 수: **{len(target_files)}개**\n\n")
        f.write("## 🗂️ [Module Index]\n\n")
        
        for file_path in sorted(target_files):
            # 🎯 [수정 4] 장부와의 대조를 위해 역슬래시 경로를 .as_posix() 정규화
            posix_rel_path = file_path.relative_to(PROJECT_ROOT).as_posix()
            f.write(f"- [{file_path.name}](./../{posix_rel_path})\n")
            
        f.write("\n## 💀 [Skeleton & Dependency 명세서]\n\n")
        for file_path in sorted(target_files):
            posix_rel_path = file_path.relative_to(PROJECT_ROOT).as_posix()
            symbols, imports, g_vars = parse_python_file(file_path)
            
            f.write(f"### 📄 파일: `{posix_rel_path}`\n")
            if imports:
                f.write(f"- **의존성(Imports):** `{', '.join(imports)}`\n")
            if g_vars:
                f.write(f"- **전역상수/변수:** `{', '.join(g_vars)}`\n")
            if symbols:
                f.write("- **내부 심볼 트리:**\n")
                for s in symbols:
                    f.write(f"  - [{s['type'].upper()}] `{s['name']}` (Line: {s['range'][0]} ~ {s['range'][1]}) -> {s['details']}\n")
            f.write("\n" + "-"*70 + "\n\n")
            
    print(f"🎯 [마스터 공장] 'CODEBASE_MAP.md'가 {len(target_files)}개의 정보로 정상 빌드되었습니다.")

def generate_ai_optimized_map():
    """AI용 정밀 요약 지도를 굽는 함수"""
    # ======================================================================
    # 🎯 [수정 2] SCAN_MODE 스위치에 맞춰 수색 대상(scan_target) 유연화
    # ======================================================================
    if SCAN_MODE == "ROOT":
        scan_target = PROJECT_ROOT
    else:
        scan_target = PROJECT_ROOT / "src"

    if not scan_target.exists():
        print(f"⚠️ {scan_target} 경로가 존재하지 않아 AI 지도 출력을 스킵합니다.")
        return

    # 프로젝트 내 파이썬 파일 수집
    target_files = []
    excludes = [".venv", ".git", "__pycache__", "system_memory", "system_maps"]
    
    for root, dirs, files in os.walk(scan_target, followlinks=True):
        # 🎯 [수정 3] 경로 구분자 리눅스식(/) 정규화 적용 (오인식 차단)
        normalized_root = root.replace("\\", "/")
        if any(kw in normalized_root for kw in excludes):
            continue
        for file in files:
            if file.endswith(".py"):
                target_files.append(Path(root) / file)

    # 장부 파일(.json) 로드
    path_to_registry = {}
    path_to_protocol = {}
    
    if REGISTRY_JSON_PATH.exists():
        try:
            with open(REGISTRY_JSON_PATH, "r", encoding="utf-8") as f:
                reg_data = json.load(f).get("registered_entities", {})
                for entity, path_str in reg_data.items():
                    if "::" in path_str:
                        p_file, _ = path_str.split("::", 1)
                        path_to_registry.setdefault(p_file, []).append(entity)
        except Exception:
            pass

    if PROTOCOL_JSON_PATH.exists():
        try:
            with open(PROTOCOL_JSON_PATH, "r", encoding="utf-8") as f:
                proto_data = json.load(f).get("protocols", {})
                for proto_name, info in proto_data.items():
                    p_file = info.get("defined_in", "")
                    fields = info.get("fields", {})
                    path_to_protocol[p_file] = (proto_name, fields)
        except Exception:
            pass

    # 마스터 AI 지도 작성 시작
    os.makedirs(OUTPUT_FILE_PATH.parent, exist_ok=True)
    with open(OUTPUT_FILE_PATH, "w", encoding="utf-8") as f:
        f.write("# 🏗️ AI-OPTIMIZED ULTRA COMPACT CODEBASE MAP (INTELLIGENT SCAN)\n\n")
        f.write("> **[AI 프로토콜 매뉴얼]** 이 문서는 파일마다 **실제 하드디스크 상대 경로 `[📂 실제경로]`**를 강제 명시해 둔 특수 지도입니다.\n")
        f.write("> AI 비서는 경로를 추측하지 말고 파일명 뒤에 박혀있는 `[📂 실제경로]` 규격을 그대로 복사하십시오.\n\n")
        f.write("```markdown\nproject_root/\n")
        
        for file_path in sorted(target_files):
            # ======================================================================
            # 🎯 [수정 4] 윈도우 역슬래시(\) 경로 에러 해결선
            # 원래는 str(file_path.relative_to(PROJECT_ROOT)) 여서 윈도우에서 역슬래시로 매칭 폭사하던 것을
            # .as_posix() 정규화를 통해 리눅스식(/) 슬래시로 변경하여 .json 장부와 100% 동기화시켰습니다.
            # ======================================================================
            posix_rel_path = file_path.relative_to(PROJECT_ROOT).as_posix()
            file_name = file_path.name
            
            # 깊이(indent) 계산
            depth = len(posix_rel_path.split("/")) - 1
            indent = "    " * depth
            
            if file_name == "start.py":
                f.write(f"{indent}├── {file_name} 🔥 [LAUNCHPAD] [📂 {posix_rel_path}]\n")
            else:
                f.write(f"{indent}├── {file_name} [📂 {posix_rel_path}]\n")
            
            # 장부 데이터 일치 검사 및 결합 (형식이 완전히 포식스로 일치하므로 매칭 성공)
            if posix_rel_path in path_to_registry:
                for reg_const in path_to_registry[posix_rel_path]:
                    f.write(f"{indent}│     ├── 🔑 [REGISTRY]: \"{reg_const}\"\n")
                    
            if posix_rel_path in path_to_protocol:
                proto_name, fields = path_to_protocol[posix_rel_path]
                f.write(f"{indent}│     ├── 📊 [PROTOCOL]: \"{proto_name}\"\n")
                field_items = [f"{k}({v.replace(' (기본값: ', ':').replace(')', '')})" for k, v in fields.items()]
                chunks = [field_items[x:x+4] for x in range(0, len(field_items), 4)]
                for chunk in chunks:
                    f.write(f"{indent}│     │     ├── {', '.join(chunk)}\n")
                    
        f.write("```\n")
        
    print(f"🎯 [마스터 공장] 'AI_CODEBASE_MAP.md'가 {len(target_files)}개의 규격으로 'system_maps'에 정상 빌드되었습니다.")

if __name__ == "__main__":
    # 백그라운드 워처 혹은 start.py 진입 시 자동 독립 가동
    generate_codebase_map()
    generate_ai_optimized_map()
